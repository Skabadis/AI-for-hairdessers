# setup_commands.txt

# 1. Connectez-vous à votre instance EC2 avec la bonne adresse IP de l'instance après le @
ssh -i ~/Desktop/'Projet Isma'/AI-for-hairdessers/skabadis_aws_ec2.pem ubuntu@13.53.120.94

# 2. Naviguez vers le répertoire de votre projet
cd ~/AI-for-hairdessers

# 3. Activez votre environnement virtuel
source venv/bin/activate

# 4. Installer les requirements
pip install -r requirements.txt

# 5. Démarrez Gunicorn
gunicorn --bind 0.0.0.0:8000 wsgi:app

# 6. Si ça ne marche pas, redémarrez gunicorn
sudo systemctl daemon-reload
sudo systemctl restart gunicorn
sudo systemctl restart nginx

# 7. Regardez le journal des logs gunicorn
sudo journalctl -u gunicorn -f
